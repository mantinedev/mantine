## Controlled

```tsx
import { useState } from 'react';
import { Textarea } from '@mantine/core';

function Demo() {
  const [value, setValue] = useState('');
  return <Textarea value={value} onChange={(event) => setValue(event.currentTarget.value)} />;
}
```
