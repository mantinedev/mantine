## Usage

use-validated-state works similar to `useState` but checks the value if it is valid based on the `rule` function.

```tsx
const [{ lastValidValue, value, valid }, setValue] = useValidatedState(
  'valid',
  (value) => value === 'valid'
);

lastValidValue; // -> valid
value; // -> valid
valid; // -> true

setValue('invalid');

lastValidValue; // -> valid
value; // -> invalid
valid; // -> false
```

## Definition

```tsx
function useValidatedState<T>(
  initialValue: T,
  validation: (value: T) => boolean
): readonly [
  {
    readonly value: T;
    readonly lastValidValue: T;
    readonly valid: boolean;
  },
  (val: T) => void
];
```
